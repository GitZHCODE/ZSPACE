159

161 
162  inMesh.vertices = new zVertex[vertices.size() * 2];
163  inMesh.edges = new zEdge[halfedges.size() * 2];
164  inMesh.faces = new zFace[faces.size() * 2];
165 
166  inMesh.setNumVertices(vertices.size());
167  inMesh.setNumEdges(floor(halfedges.size()));
168  inMesh.setNumPolygons(faces.size());
169 
170  inMesh.vertexActive.clear();

172  {
173  inMesh.vertices[i].setVertexId(i);
174  if (vertices[i] != -1) inMesh.vertices[i].setEdge(&inMesh.edges[vertices[i]]);
175 
176  inMesh.vertexActive.push_back(true);
177  }
178 
179  int k = 0;
180  inMesh.edgeActive.clear();
181  for (int i = 0; i < halfedges.size(); i++)
182  {
183  inMesh.edges[i].setEdgeId(i);
184 
185  if (halfedges[i][k] != -1) inMesh.edges[i].setPrev(&inMesh.edges[halfedges[i][k]]);
186  if (halfedges[i][k + 1] != -1) inMesh.edges[i].setNext(&inMesh.edges[halfedges[i][k + 1]]);
187  if (halfedges[i][k + 2] != -1) inMesh.edges[i].setVertex(&inMesh.vertices[halfedges[i][k + 2]]);
188  if (halfedges[i][k + 3] != -1) inMesh.edges[i].setFace(&inMesh.faces[halfedges[i][k + 3]]);
189 
190 
191  if (i % 2 == 0) inMesh.edges[i].setSym(&inMesh.edges[i]);
192  else inMesh.edges[i].setSym(&inMesh.edges[i - 1]);
193 
194  inMesh.edgeActive.push_back(true);
195  }
196 
197  inMesh.faceActive.clear();
198  for (int i = 0; i < faces.size(); i++)
199  {
200  inMesh.faces[i].setFaceId(i);
201  if (faces[i] != -1) inMesh.faces[i].setEdge(&inMesh.edges[faces[i]]);
202 
203  inMesh.faceActive.push_back(true);
204  }
205
