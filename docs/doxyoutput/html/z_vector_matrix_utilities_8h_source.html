284  bool pointInTriangle(zVector &pt, zVector &t0, zVector &t1, zVector &t2)
285  {
286  // Compute vectors
287  zVector v0 = t2 - t0;
288  zVector v1 = t1 - t0;
289  zVector v2 = pt - t0;
290 
291  // Compute dot products
292  double dot00 = v0 * v0;
293  double dot01 = v0 * v1;
294  double dot02 = v0 * v2;
295  double dot11 = v1 * v1;
296  double dot12 = v1 * v2;
297 
298  // Compute barycentric coordinates
299  double invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
300  double u = (dot11 * dot02 - dot01 * dot12) * invDenom;
301  double v = (dot00 * dot12 - dot01 * dot02) * invDenom;
302 
303  if (abs(u) < 0.001) u = 0;
304  if (abs(v) < 0.001) v = 0;
305 
306  // round factor to precision 3
307  double factor = pow(10, 3);
308  u = round(u*factor)/factor;
309  v = round(v*factor) / factor;
310 
311  //printf("\n u : %1.2f v: %1.2f ", u, v);
312 
313  // Check if point is in triangle
314  return ((u >= 0) && (v >= 0) && (u + v <= 1));
