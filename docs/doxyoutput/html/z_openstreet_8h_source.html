244  double computeDistance(double &lat1, double &lon1, double &lat2, double &lon2)
245  {
246  double R = 6378.137; // Radius of earth in KM
247 
248  double dLat = (lat2 * PI / 180) - (lat1 * PI / 180);
249  double dLon = (lon2 * PI / 180) - (lon1 * PI / 180);
250 
251  double a = sin(dLat / 2) * sin(dLat / 2) + cos(lat1 * PI / 180) * cos(lat2 * PI / 180) * sin(dLon / 2) * sin(dLon / 2);
252  double c = 2 * atan2(sqrt(a), sqrt(1 - a));
253  double d = R * c;
254 
255  return d; // in kilometers
256  }
