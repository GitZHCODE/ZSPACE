245  double computeDistance(double &lat1, double &lon1, double &lat2, double &lon2)
246  {
247  double R = 6378.137; // Radius of earth in KM
248 
249  double dLat = (lat2 * PI / 180) - (lat1 * PI / 180);
250  double dLon = (lon2 * PI / 180) - (lon1 * PI / 180);
251 
252  double a = sin(dLat / 2) * sin(dLat / 2) + cos(lat1 * PI / 180) * cos(lat2 * PI / 180) * sin(dLon / 2) * sin(dLon / 2);
253  double c = 2 * atan2(sqrt(a), sqrt(1 - a));
254  double d = R * c;
255 
256  return d; // in kilometers
257  }
